    const coin = 상태.코인목록[symbol];
    if (!coin) return;

    // 4개 분할 차트별로 비동기 로드 프로미스들 동시 가동
    const 로드프로미스들 = 상태.차트객체.분할차트들.map((_, idx) => {
        return 분할차트캔들데이터로드(symbol, idx);
    });
    
    await Promise.all(로드프로미스들);
    
    // 로드 완료 후 렌더링
    분할차트들렌더링();
}
